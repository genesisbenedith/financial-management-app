import java.io.File;

public class UserManager {
    private static final String USER_DATA_DIRECTORY = "data";

    // Create the data directory if it doesn't already exist
    public static void createUserDataDir() {
        File directory = new File(USER_DATA_DIRECTORY);
        if (!directory.exists()) {
            boolean created = directory.mkdir();
            if (created) {
                System.out.println("Data directory created: " + USER_DATA_DIRECTORY);
            } else {
                System.err.println("Failed to create data directory: " + USER_DATA_DIRECTORY);
            }
        }
    }

    
}
